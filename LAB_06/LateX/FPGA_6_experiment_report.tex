\subsection*{Simulation --- VGA\_Interface}

The design targets the \textbf{800$\times$600 @ \SI{72}{\hertz}} VGA mode driven by a
\SI{50}{\mega\hertz} pixel clock, generated by dividing the board's
\SI{100}{\mega\hertz} system clock by two. Two counters track the beam position:
\texttt{h\_count} runs \numrange{0}{1039} (\num{1040} pixel-clocks per line) and
\texttt{v\_count} runs \numrange{0}{665} (\num{666} lines per frame). Only the
\num{800}$\times$\num{600} window is visible; outside it the outputs are blanked to
black. % TODO: expand with your own wording

\vspace{1cm}
\subsection{Pixel-clock and Blanking Considerations}

The \SI{50}{\mega\hertz} pixel rate is produced by toggling a register
(\texttt{pclk}) on every system-clock edge, i.e.\ dividing the
\SI{100}{\mega\hertz} clock by two. The sync and colour outputs are registered
(flip-flops) so the VGA pins stay glitch-free, and outside the
\num{800}$\times$\num{600} visible window the colour outputs are forced to
\texttt{\{R,G,B\} = 0} (blanking).

\textbf{Signals} (top to bottom in the waveform):
\begin{itemize}[noitemsep,topsep=2pt,leftmargin=1.4em]
  \item \texttt{clk} - \SI{100}{\mega\hertz} system clock (test-bench driven).
  \item \texttt{rstn} - active-low reset.
  \item \texttt{pclk} - internal \SI{50}{\mega\hertz} pixel clock (clk divided by two).
  \item \texttt{pixel\_color[11:0]} - input colour word \{R,G,B\} from the Drawer.
  \item \texttt{XCoord[10:0]} / \texttt{YCoord[10:0]} - current pixel position.
  \item \texttt{Hsync} / \texttt{Vsync} - sync outputs (positive polarity).
  \item \texttt{vgaRed/Green/Blue[3:0]} - colour outputs (forced to 0 in blanking).
\end{itemize}

\section*{Problems we faced}

Generating the \SI{50}{\mega\hertz} pixel clock by dividing the system clock in
fabric (the \texttt{pclk} register) caused Vivado to flag every register it drives
as a \emph{non-clocked sequential cell} (TIMING-17), because a fabric-divided clock
is not a recognised timing clock. Declaring it with
\texttt{create\_generated\_clock} (alternatively, using a clock-enable instead of a
second clock) makes those paths timed and clears the warnings.

\subsection*{VGA\_Interface.v}
\begin{lstlisting}
`timescale 1ns/10ps
// Module Name: VGA_Interface - VGA timing/sync generator

`define INC(x, max) ((x == max-1)? 0 : (x + 1))

module VGA_Interface(
    input  wire        clk,
    input  wire        rstn,
    input  wire [11:0] pixel_color,
    output reg  [3:0]  vgaRed,
    output reg  [3:0]  vgaGreen,
    output reg  [3:0]  vgaBlue,
    output reg         Hsync,
    output reg         Vsync,
    output wire [10:0] XCoord,
    output wire [10:0] YCoord
    );

    reg [10:0] h_count;
    reg [10:0] v_count;

    localparam H_VISIBLE = 799, H_FRONT_PORCH = 855, H_SYNC_END = 975, H_TOTAL = 1039;
    localparam V_VISIBLE = 599, V_FRONT_PORCH = 636, V_SYNC_END = 642, V_TOTAL = 665;

    // 50 MHz pixel-clock (divide 100 MHz by 2)
    reg pclk = 0;
    always @(posedge clk) pclk <= ~pclk;

    // Display logic (registered outputs)
    always @(posedge pclk) begin
        if (!rstn) begin
            Hsync <= 1'b0; Vsync <= 1'b0;
            vgaRed <= 4'h0; vgaGreen <= 4'h0; vgaBlue <= 4'h0;
        end else begin
            Hsync <= (h_count > H_FRONT_PORCH) && (h_count <= H_SYNC_END);
            Vsync <= (v_count > V_FRONT_PORCH) && (v_count <= V_SYNC_END);
            if ((h_count <= H_VISIBLE) && (v_count <= V_VISIBLE)) begin
                vgaRed   <= pixel_color[11:8];
                vgaGreen <= pixel_color[7:4];
                vgaBlue  <= pixel_color[3:0];
            end else begin
                vgaRed <= 4'h0; vgaGreen <= 4'h0; vgaBlue <= 4'h0;
            end
        end
    end

    // Coordinate counters
    always @(posedge pclk) begin
        if (!rstn) begin
            h_count <= 0; v_count <= 0;
        end else begin
            h_count <= `INC(h_count, H_TOTAL+1);
            if (h_count == H_TOTAL)
                v_count <= `INC(v_count, V_TOTAL+1);
        end
    end

    assign XCoord = h_count;
    assign YCoord = v_count;

endmodule
\end{lstlisting}

\subsection*{Drawer.v}
\begin{lstlisting}
`timescale 1ns/10ps
// Module Name: Drawer - pixel color / movable rectangle logic

module Drawer(
    input  wire        clk,
    input  wire        reset,
    input  wire [11:0] sw,
    input  wire        btnu,
    input  wire        btnd,
    input  wire        btnl,
    input  wire        btnr,
    input  wire [10:0] XCoord,
    input  wire [10:0] YCoord,
    output wire [11:0] pixel_color
    );

    // 800x600 visible region downsampled by 8
    localparam integer GRID_W = 100;
    localparam integer GRID_H = 75;
    localparam integer CX     = GRID_W / 2; // 50
    localparam integer CY     = GRID_H / 2; // 37

    // 20-bit debouncer
    wire btnu_pulse, btnd_pulse, btnl_pulse, btnr_pulse;
    Debouncer #(.COUNTER_BITS(20)) db_u (.clk(clk), .input_unstable(btnu), .output_stable(btnu_pulse));
    Debouncer #(.COUNTER_BITS(20)) db_d (.clk(clk), .input_unstable(btnd), .output_stable(btnd_pulse));
    Debouncer #(.COUNTER_BITS(20)) db_l (.clk(clk), .input_unstable(btnl), .output_stable(btnl_pulse));
    Debouncer #(.COUNTER_BITS(20)) db_r (.clk(clk), .input_unstable(btnr), .output_stable(btnr_pulse));

    // Rectangle edges in coarse-grid units (inclusive)
    reg [7:0] top, bottom, left, right;

    always @(posedge clk) begin
        if (reset) begin
            top <= CY; bottom <= CY; left <= CX; right <= CX;
        end else begin
            if (btnu_pulse && top    >  0          ) top    <= top    - 1'b1;
            if (btnd_pulse && bottom <  GRID_H - 1 ) bottom <= bottom + 1'b1;
            if (btnl_pulse && left   >  0          ) left   <= left   - 1'b1;
            if (btnr_pulse && right  <  GRID_W - 1 ) right  <= right  + 1'b1;
        end
    end

    // Coarse-grid coordinate (divide by 8)
    wire [7:0] x_ds = XCoord[10:3];
    wire [7:0] y_ds = YCoord[10:3];

    wire inside = (x_ds >= left) && (x_ds <= right)
               && (y_ds >= top ) && (y_ds <= bottom);

    assign pixel_color = inside ? sw : 12'h000;

endmodule
\end{lstlisting}

\subsection*{Debouncer.v}
\begin{lstlisting}
`timescale 1ns/10ps
// Module Name: Debouncer - one clean 1-cycle pulse per stable button press

module Debouncer(clk, input_unstable, output_stable);

   input clk, input_unstable;
   output reg output_stable;

   parameter COUNTER_BITS = 7;
   reg prev_msb;
   reg [COUNTER_BITS-1:0] counter; // Hysteresis counter

   always @(posedge clk) begin
        if (input_unstable == 1)
            counter <= (counter < {COUNTER_BITS{1'b1}}) ? counter + 1 : counter;
        else
            counter <= (counter > {COUNTER_BITS{1'b0}}) ? counter - 1 : counter;

        // 1-cycle pulse on the 0->1 transition of the counter MSB
        prev_msb      <= counter[COUNTER_BITS-1];
        output_stable <= (~prev_msb) & counter[COUNTER_BITS-1];
   end

endmodule
\end{lstlisting}
